#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"

#define GREEN_LED  GPIO_NUM_22
#define BLUE_LED   GPIO_NUM_27

// Task for GREEN LED
void green_led_task(void *pvParameters)
{
    while (1) {
        printf("GREEN ON\n");
        gpio_set_level(GREEN_LED, 1);
        vTaskDelay(pdMS_TO_TICKS(500));

        printf("GREEN OFF\n");
        gpio_set_level(GREEN_LED, 0);
        vTaskDelay(pdMS_TO_TICKS(500));
    }
}

// Task for BLUE LED
void blue_led_task(void *pvParameters)
{
    while (1) {
        printf("BLUE ON\n");
        gpio_set_level(BLUE_LED, 1);
        vTaskDelay(pdMS_TO_TICKS(1000));

        printf("BLUE OFF\n");
        gpio_set_level(BLUE_LED, 0);
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

void app_main(void)
{
    // Configure GPIOs
    gpio_reset_pin(GREEN_LED);
    gpio_reset_pin(BLUE_LED);

    gpio_set_direction(GREEN_LED, GPIO_MODE_OUTPUT);
    gpio_set_direction(BLUE_LED, GPIO_MODE_OUTPUT);

    // Create tasks
    xTaskCreate(green_led_task, "Green LED Task", 2048, NULL, 1, NULL);
    xTaskCreate(blue_led_task,  "Blue LED Task",  2048, NULL, 1, NULL);
}
